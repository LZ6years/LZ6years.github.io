git config -l                           # 查看当前生效的所有 Git 配置（多层级汇总）  
git config --system --list              # 查看系统级配置（针对整台电脑）  
git config --global --list              # 查看用户级配置（当前用户）  
git config --local  --list              # 查看当前仓库的本地配置（需在仓库目录中执行）  
​  
git config --global user.name "Your Name"     # 设置全局用户名  
git config --global user.email "you@example.com" # 设置全局邮箱  
git config --global init.defaultBranch main   # 设置新建仓库默认分支为 main  
​
git config --global core.ignorecase false     # 让 Git 区分大小写文件名  
git config --global pull.rebase false         # pull 默认使用 merge，可改为 true 变为 rebase  
git config --global merge.ff only             # 禁止非快进合并  
git config --global fetch.prune true          # 拉取时自动清理已删除的远程分支

git init                                # 在当前目录初始化一个新的 Git 仓库  
git clone <url>                         # 从 url 的远程地址克隆整个仓库到本地  
git clone --depth=1 <url>               # 从 url 克隆最近一次提交（浅克隆），加快速度  
git clone --branch <branch> <url>       # 从 url 克隆指定分支 <branch> 的仓库  
​  
git status                                   # 显示工作区、暂存区状态  
git diff                                     # 比较工作区与暂存区的差异  
git diff --staged                            # 比较暂存区与最新提交的差异  
git show <commit>                            # 查看指定提交的详细内容  
​  
git add <file>                               # 将指定文件加入暂存区  
git add -A                                   # 将所有修改、删除、创建文件加入暂存区  
git restore --staged <file>                  # 从暂存区移除文件（取消暂存）  
git commit -m "提交信息"                     # 创建一个新提交  
git commit --amend                           # 修改最近一次提交（信息或包含新改动）  
​  
git mv <old> <new>          # 重命名文件并记录到版本历史  
git rm <file>               # 删除文件并记录到版本历史(工作区和暂存区同时删除)
git rm --cacged <file>      # 删除文件并记录到版本历史(删除暂存区文件，不删工作区)

git branch                                   # 查看所有本地分支  
git branch -vv                               # 查看分支及其上游分支与最近提交  
git branch -d <branch>                       # 删除已经合并的分支  
git branch -D <branch>                       # 强制删除分支  
​  
git switch <branch>                          # 切换到已存在的分支  
git switch -c <new-branch>                   # 创建并切换到新分支 <new-branch>  

git merge <branch>                           # 将指定分支合并到当前分支  
git merge --no-ff <branch>                   # 禁用快进合并，保留合并提交  
​  
git rebase <base-branch>                     # 将当前分支变基到指定分支之上  
git rebase -i <base>                         # 交互式变基，可编辑、压缩提交  
git rebase --abort                           # 取消正在进行的变基操作  
git rebase --continue                        # 解决冲突后继续变基  
​  
git cherry-pick <commit>                     # 将某次提交应用到当前分支  
git revert <commit>                          # 生成一个新的提交来撤销指定提交

git remote -v                                # 查看所有已配置的远程仓库地址  
git remote add origin <url>                  # 添加名为 origin 的远程仓库  
git remote remove origin                     # 删除名为 origin 的远程仓库  
git remote set-url origin <new-url>          # 修改 origin 的远程地址  
​  
git fetch                               # 从远程拉取最新引用和对象，不自动合并  
git fetch -p                            # 拉取同时清理已删除的远程分支引用  
git pull                                # 相当于 fetch + merge  
git pull --rebase                       # 拉取后用 rebase 合并，保持线性历史  
​  
git push                                     # 推送当前分支到远程  
git push -u origin <branch>                  # 推送当前分支并设置上游  
git push origin --delete <branch>            # 删除远程分支  
git push --tags                              # 推送本地所有标签到远程

git reset --soft <commit>      # 回退到指定提交，保留工作区和暂存区  
git reset --mixed <commit>     # 回退到指定提交，保留工作区，清空暂存区

git restore --source=<commit> <file>         # 用指定提交的内容覆盖工作区文件  
git restore --staged <file>                  # 从暂存区移除文件（取消暂存）

git stash push -m "说明"                     # 暂存当前所有更改到一个命名的 stash  
git stash push -u -m "说明"                  # 包括未跟踪文件一起暂存  
git stash list                               # 查看 stash 列表  
git stash show -p stash@{0}                  # 查看指定 stash 的改动详情  
git stash pop                                # 恢复最近一个 stash 并删除它  
git stash apply stash@{n}                    # 恢复指定 stash 但保留它  
git stash drop stash@{n}                     # 删除指定 stash  
git stash clear                              # 删除所有 stash（危险操作）

git init                                     # 初始化仓库  
git add -A                                   # 添加所有改动到暂存区  
git commit -m "chore: initial commit"        # 首次提交  
git branch -M main                           # 重命名分支为 main  
git remote add origin <url>                  # 绑定远程仓库  
git push -u origin main                      # 推送 main 分支并建立跟踪关系  
​  
git switch -c feature/login                  # 新建功能分支并切换  
git add -A                                   # 提交前添加所有修改  
git commit -m "feat(login): basic form"      # 提交功能代码  
git push -u origin feature/login             # 推送到远程  
​  
git switch feature/login                     # 切回功能分支  
git fetch origin                             # 拉取远程更新  
git rebase origin/main                       # 将分支变基到最新 main  
git push --force-with-lease                  # 强制推送（安全模式）  
​  
git switch main                              # 切回主分支  
git pull                                     # 更新主分支  
git tag -a v1.2.0 -m "Release v1.2.0"        # 打标签发布  
git push origin v1.2.0                       # 推送标签到远程

10、冲突解决 

1
2
3
4
5

git add <冲突文件>                           # 解决冲突后标记文件为已解决  
git merge --continue                         # 继续合并  
git rebase --continue                        # 继续变基  
git merge --abort                            # 放弃合并回到合并前状态  
git rebase --abort                           # 放弃变基回到开始前状态

11、找回丢失提交 

1
2
3

git reflog                                   # 找到需要回退的提交或 HEAD@{n}  
git reset --hard <that-commit>               # 回到该提交（危险，丢弃未提交改动）  
git revert <that-commit>                     # 创建一个新的提交反做该提交
